django-admin - list of commands django
django-admin startproject "projectname" - create project

python manage.py startapp "appname"
python manage.py runserver - run serveer
ctrl/c - close server

python manage.py makemigrations
python manage.py migrate


#fixtures
python manage.py dumpdata "класс таблицы" > "расположение куда сохранить"
пример - python manage.py dumpdata catalog.Categories > fixtures/catalog/cats.json

python manage.py loaddata ".json"
пример - python manage.py loaddata fixtures/catalog/cats.json
